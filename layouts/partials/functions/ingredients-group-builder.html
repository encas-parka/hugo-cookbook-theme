from ingredients index quantite, // scaled if mode = "evenement" unit, // unit as declared on the item commentaire, // free text } Notes: - Unresolved ingredients go under "absent". - Type is read := .quantite | default 0 -}} {{- $u := .unit | default "" -}} {{- $c := .commentaire -}} {{- $meta := dict -}} {{- if $ingId -}} {{- $uuid := partial "functions/extract-uuid-from-slug" $ingId -}} {{- (index $grouped "absent") | default (slice) -}} {{- $grouped = merge $grouped (dict "absent" ( $current | append $entry )) -}} {{- end -}} {{- end -}} {{- return $grouped -}}
